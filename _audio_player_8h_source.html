94 inline std::string playerStateToString(AudioPlayerState state) {
95  switch (state) {
96  case AudioPlayerState::IDLE:
97  return "IDLE";
98  case AudioPlayerState::PLAYING:
99  return "PLAYING";
100  case AudioPlayerState::STOPPED:
101  return "STOPPED";
102  case AudioPlayerState::PAUSED:
103  return "PAUSED";
104  case AudioPlayerState::BUFFERING:
105  return "BUFFERING";
106  case AudioPlayerState::BUFFER_UNDERRUN:
107  return "BUFFER_UNDERRUN";
108  case AudioPlayerState::FINISHED:
109  return "FINISHED";
110  }
111  return "unknown AudioPlayerState";
112 }
