357 template <typename ParamType>
358 bool CurlEasyHandleWrapper::setopt(CURLoption option, ParamType value) {
359  auto result = curl_easy_setopt(m_handle, option, value);
360  if (result != CURLE_OK) {
361  logger::acsdkError(logger::LogEntry("CurlEasyHandleWrapper", "setoptFailed")
362  .d("reason", "curl_easy_setopt failed")
363  .d("option", option)
364  .sensitive("value", value)
365  .d("result", result)
366  .d("error", curl_easy_strerror(result)));
367  return false;
368  }
369  return true;
370 }
