154 template <typename Task, typename... Args>
155 auto TaskQueue::pushTo(bool front, Task task, Args&&... args) -> std::future<decltype(task(args...))> {
156  // Remove arguments from the tasks type by binding the arguments to the task.
157  auto boundTask = std::bind(std::forward<Task>(task), std::forward<Args>(args)...);
158 
159  /*
160  * Create a std::packaged_task with the correct return type. The decltype only returns the return value of the
161  * boundTask. The following parentheses make it a function call with the boundTask return type. The package task
162  * will then return a future of the correct type.
163  *
164  * Note: A std::packaged_task fulfills its future *during* the call to operator(). If the user of a
165  * std::packaged_task hands it off to another thread to execute, and then waits on the future, they will be able to
166  * retrieve the return value from the task and know that the task has executed, but they do not know exactly when
167  * the task object has been deleted. This distinction can be significant if the packaged task is holding onto
168  * resources that need to be freed (through a std::shared_ptr for example). If the user needs to wait for those
169  * resources to be freed they have no way of knowing how long to wait. The translated_task lambda below is a
170  * workaround for this limitation. It executes the packaged task, then disposes of it before passing the task's
171  * return value back to the future that the user is waiting on.
172  */
173  using PackagedTaskType = std::packaged_task<decltype(boundTask())()>;
174  auto packaged_task = std::make_shared<PackagedTaskType>(boundTask);
175 
176  // Create a promise/future that we will fulfill when we have cleaned up the task.
177  auto cleanupPromise = std::make_shared<std::promise<decltype(task(args...))>>();
178  auto cleanupFuture = cleanupPromise->get_future();
179 
180  // Remove the return type from the task by wrapping it in a lambda with no return value.
181  auto translated_task = [packaged_task, cleanupPromise]() mutable {
182  // Execute the task.
183  packaged_task->operator()();
184  // Note the future for the task's result.
185  auto taskFuture = packaged_task->get_future();
186  // Clean up the task.
187  packaged_task.reset();
188  // Forward the task's result to our cleanup promise/future.
189  forwardPromise(cleanupPromise, &taskFuture);
190  };
191 
192  // Release our local reference to packaged task so that the only remaining reference is inside the lambda.
193  packaged_task.reset();
194 
195  {
196  std::lock_guard<std::mutex> queueLock{m_queueMutex};
197  if (!m_shutdown) {
198  m_queue.emplace(front ? m_queue.begin() : m_queue.end(), new std::function<void()>(translated_task));
199  } else {
200  using FutureType = decltype(task(args...));
201  return std::future<FutureType>();
202  }
203  }
204 
205  m_queueChanged.notify_all();
206  return cleanupFuture;
207 }
